#### 4.1.2 规则解析实现

```java
@Slf4j
@Service
@RequiredArgsConstructor
public class SignPointsCalculator {

    private final SysConfigService configService;
    
    private volatile List<DailyPointsRule> cachedRules;
    private volatile long cacheExpireTime;

    @Data
    public static class DailyPointsRule {
        private Integer minDays;
        private Integer maxDays;
        private Integer points;
    }

    public int getBasePoints(int continuousDays) {
        List<DailyPointsRule> rules = getRules();
        return rules.stream()
                .filter(r -> continuousDays >= r.getMinDays() && continuousDays <= r.getMaxDays())
                .findFirst()
                .map(DailyPointsRule::getPoints)
                .orElse(5);
    }

    private List<DailyPointsRule> getRules() {
        if (cachedRules != null && System.currentTimeMillis() < cacheExpireTime) {
            return cachedRules;
        }
        
        synchronized (this) {
            if (cachedRules != null && System.currentTimeMillis() < cacheExpireTime) {
                return cachedRules;
            }
            
            String json = configService.getValue("sign.daily_points_rules");
            cachedRules = JSON.parseArray(json, DailyPointsRule.class);
            cacheExpireTime = System.currentTimeMillis() + 3600000;
            return cachedRules;
        }
    }
}
```

#### 4.2.3 奖励发放实现

```java
@Slf4j
@Service
@RequiredArgsConstructor
public class SignRewardService {

    private final SignRewardRuleMapper rewardRuleMapper;
    private final SignRewardRecordMapper rewardRecordMapper;
    private final UserPointsService userPointsService;
    private final VoucherService voucherService;
    private final LotteryService lotteryService;

    @Transactional(rollbackFor = Exception.class)
    public ClaimResultVO claimReward(Long userId, Long ruleId) {
        String lockKey = "lock:sign:claim:" + userId + ":" + ruleId;
        
        return redisLockUtil.executeWithLock(lockKey, 5, TimeUnit.SECONDS, () -> {
            SignRewardRule rule = rewardRuleMapper.selectById(ruleId);
            if (rule == null || rule.getStatus() != 1) {
                throw new BizException(40032, "奖励活动已结束");
            }

            String cycleMonth = getCycleMonth();
            boolean claimed = rewardRecordMapper.existsByUserRuleCycle(userId, ruleId, cycleMonth);
            if (claimed) {
                throw new BizException(40031, "该奖励已领取");
            }

            Object rewardDetail = grantReward(userId, rule);

            SignRewardRecord record = new SignRewardRecord();
            record.setUserId(userId);
            record.setRuleId(ruleId);
            record.setCycleMonth(cycleMonth);
            record.setRewardType(rule.getRewardType());
            record.setRewardValue(rule.getRewardValue());
            record.setStatus(1);
            rewardRecordMapper.insert(record);

            return ClaimResultVO.builder()
                    .ruleId(ruleId)
                    .rewardType(rule.getRewardType())
                    .rewardName(rule.getRewardName())
                    .rewardDetail(rewardDetail)
                    .build();
        });
    }

    private Object grantReward(Long userId, SignRewardRule rule) {
        return switch (rule.getRewardType()) {
            case 1 -> grantLottery(userId, rule);
            case 2 -> grantVoucher(userId, rule);
            case 3 -> grantPoints(userId, rule);
            default -> throw new BizException(50000, "未知奖励类型");
        };
    }

    private Object grantLottery(Long userId, SignRewardRule rule) {
        lotteryService.grantChance(userId, rule.getRewardValue());
        return Map.of("lotteryChances", rule.getRewardValue());
    }

    private Object grantVoucher(Long userId, SignRewardRule rule) {
        VoucherGrantResult result = voucherService.grantSignRewardVoucher(userId, rule.getVoucherId());
        return Map.of(
            "voucherId", result.getVoucherId(),
            "voucherTitle", result.getTitle()
        );
    }

    private Object grantPoints(Long userId, SignRewardRule rule) {
        userPointsService.addPoints(userId, rule.getRewardValue(), 
            "SIGN_MILESTONE", rule.getId(), "签到里程碑奖励");
        return Map.of("points", rule.getRewardValue());
    }
}
```

### 4.3 积分上限控制

#### 4.3.1 单日积分上限

```java
@Service
@RequiredArgsConstructor
public class PointsLimitService {

    private final StringRedisTemplate redisTemplate;

    private static final int DAILY_POINTS_LIMIT = 500;

    public boolean checkDailyLimit(Long userId, int addPoints) {
        String key = "points:daily:" + userId + ":" + LocalDate.now();
        String value = redisTemplate.opsForValue().get(key);
        int currentDaily = value == null ? 0 : Integer.parseInt(value);
        
        if (currentDaily + addPoints > DAILY_POINTS_LIMIT) {
            return false;
        }
        
        redisTemplate.opsForValue().increment(key, addPoints);
        redisTemplate.expire(key, Duration.ofDays(1));
        return true;
    }
}
```

#### 4.3.2 积分余额上限

```java
public class PointsConstants {
    public static final int MAX_POINTS_BALANCE = 100000;
}

@Service
public class UserPointsServiceImpl implements UserPointsService {

    @Override
    @Transactional(rollbackFor = Exception.class)
    public boolean addPoints(Long userId, Integer points, String bizType, Long bizId, String description) {
        UserInfo userInfo = userInfoMapper.selectById(userId);
        int newBalance = userInfo.getPoints() + points;
        
        if (newBalance > PointsConstants.MAX_POINTS_BALANCE) {
            newBalance = PointsConstants.MAX_POINTS_BALANCE;
            points = PointsConstants.MAX_POINTS_BALANCE - userInfo.getPoints();
            log.warn("用户积分达到上限: userId={}, capped at {}", userId, PointsConstants.MAX_POINTS_BALANCE);
        }
        
        userInfoMapper.updatePoints(userId, newBalance);
        savePointsRecord(userId, points, newBalance, bizType, bizId, description);
        return true;
    }
}
```

### 5.1 分布式锁机制

```java
@Slf4j
@Component
@RequiredArgsConstructor
public class RedisLockUtil {

    private final StringRedisTemplate redisTemplate;

    public boolean tryLock(String key, long expire, TimeUnit unit) {
        String value = UUID.randomUUID().toString();
        Boolean success = redisTemplate.opsForValue()
                .setIfAbsent(key, value, expire, unit);
        return Boolean.TRUE.equals(success);
    }

    public void unlock(String key) {
        String script = "if redis.call('get', KEYS[1]) == ARGV[1] then " +
                        "return redis.call('del', KEYS[1]) " +
                        "else return 0 end";
        redisTemplate.execute(
            new DefaultRedisScript<>(script, Long.class),
            Collections.singletonList(key),
            redisTemplate.opsForValue().get(key)
        );
    }

    public <T> T executeWithLock(String key, long expire, TimeUnit unit, Supplier<T> supplier) {
        if (!tryLock(key, expire, unit)) {
            throw new BizException(50001, "操作频繁，请稍后重试");
        }
        try {
            return supplier.get();
        } finally {
            unlock(key);
        }
    }
}
```

### 5.3 事务一致性

```java
@Slf4j
@Service
@RequiredArgsConstructor
public class SignServiceImpl implements SignService {

    private final SignMapper signMapper;
    private final UserPointsService userPointsService;
    private final RedisLockUtil redisLockUtil;

    private static final ZoneId ZONE = ZoneId.of("Asia/Shanghai");

    @Override
    @Transactional(rollbackFor = Exception.class)
    public SignResultVO doSign(Long userId) {
        String lockKey = "lock:sign:" + userId;
        
        return redisLockUtil.executeWithLock(lockKey, 5, TimeUnit.SECONDS, () -> {
            LocalDate today = LocalDate.now(ZONE);
            
            Sign sign = getOrCreateSign(userId, today.getYear(), today.getMonthValue());
            
            if (isSigned(sign.getSignBitmap(), today.getDayOfMonth())) {
                throw new BizException(40020, "今日已签到");
            }
            
            int continuousDays = calcContinuousDays(userId, today);
            int basePoints = signPointsCalculator.getBasePoints(continuousDays);
            int bonusPoints = signPointsCalculator.getBonusPoints(continuousDays);
            int totalPoints = basePoints + bonusPoints;
            
            int newBitmap = sign.getSignBitmap() | (1 << (today.getDayOfMonth() - 1));
            signMapper.updateBitmapWithCAS(sign.getId(), newBitmap, sign.getSignDays() + 1, sign.getSignBitmap());
            
            userPointsService.addPoints(userId, totalPoints, "SIGN", sign.getId(), 
                buildSignDesc(basePoints, bonusPoints, continuousDays));
            
            updateContinuousDaysCache(userId, continuousDays);
            
            return buildResult(today, basePoints, bonusPoints, continuousDays);
        });
    }
}
```

#### 5.4.2 补偿机制

```java
@Slf4j
@Service
@RequiredArgsConstructor
public class SignCompensationService {

    private final SignMapper signMapper;
    private final UserInfoMapper userInfoMapper;
    private final PointsRecordMapper pointsRecordMapper;

    @Scheduled(cron = "0 0 2 * * ?")
    public void reconcileSignPoints() {
        log.info("开始执行签到积分对账任务");
        
        LocalDate yesterday = LocalDate.now().minusDays(1);
        List<Sign> signs = signMapper.selectByDate(yesterday);
        
        for (Sign sign : signs) {
            try {
                reconcileUser(sign);
            } catch (Exception e) {
                log.error("对账失败: userId={}", sign.getUserId(), e);
            }
        }
    }

    private void reconcileUser(Sign sign) {
        int expectedPoints = calculateExpectedPoints(sign);
        int actualPoints = pointsRecordMapper.sumPointsByUserAndBiz(sign.getUserId(), "SIGN", sign.getId());
        
        if (expectedPoints != actualPoints) {
            int diff = expectedPoints - actualPoints;
            if (diff > 0) {
                userInfoMapper.addPoints(sign.getUserId(), diff);
                log.info("积分补偿: userId={}, points={}", sign.getUserId(), diff);
            }
        }
    }
}
```

### 5.5 缓存一致性

```java
@Slf4j
@Service
@RequiredArgsConstructor
public class SignCacheService {

    private final StringRedisTemplate redisTemplate;

    private static final String BITMAP_KEY_PREFIX = "sign:bitmap:";
    private static final String CONTINUOUS_KEY_PREFIX = "sign:continuous:";

    public void updateSignCache(Long userId, int year, int month, int bitmap, int continuousDays) {
        String bitmapKey = BITMAP_KEY_PREFIX + userId + ":" + year + ":" + month;
        String continuousKey = CONTINUOUS_KEY_PREFIX + userId;
        
        LocalDate monthEnd = LocalDate.of(year, month, 1).plusMonths(1).plusDays(6);
        long ttl = Duration.between(LocalDateTime.now(), monthEnd.atStartOfDay()).getSeconds();
        
        redisTemplate.opsForValue().set(bitmapKey, String.valueOf(bitmap), 
            Math.max(ttl, 86400), TimeUnit.SECONDS);
        redisTemplate.opsForValue().set(continuousKey, String.valueOf(continuousDays), 
            48, TimeUnit.HOURS);
    }

    public void evictSignCache(Long userId, int year, int month) {
        String bitmapKey = BITMAP_KEY_PREFIX + userId + ":" + year + ":" + month;
        redisTemplate.delete(bitmapKey);
    }

    public Integer getSignBitmap(Long userId, int year, int month) {
        String key = BITMAP_KEY_PREFIX + userId + ":" + year + ":" + month;
        String value = redisTemplate.opsForValue().get(key);
        return value == null ? null : Integer.parseInt(value);
    }
}
```

```java
@Slf4j
@SpringBootTest
class SignConcurrencyTest {

    @Autowired
    private SignService signService;

    @Test
    @DisplayName("并发签到 - 应只有一个成功")
    void testConcurrentSign() throws InterruptedException {
        Long userId = 999999L;
        int threadCount = 10;
        CountDownLatch startLatch = new CountDownLatch(1);
        CountDownLatch endLatch = new CountDownLatch(threadCount);
        AtomicInteger successCount = new AtomicInteger(0);
        AtomicInteger failCount = new AtomicInteger(0);

        ExecutorService executor = Executors.newFixedThreadPool(threadCount);

        for (int i = 0; i < threadCount; i++) {
            executor.submit(() -> {
                try {
                    startLatch.await();
                    signService.doSign(userId);
                    successCount.incrementAndGet();
                } catch (BizException e) {
                    if (e.getCode() == 40020) {
                        failCount.incrementAndGet();
                    }
                } catch (Exception ignored) {
                } finally {
                    endLatch.countDown();
                }
            });
        }

        startLatch.countDown();
        endLatch.await(10, TimeUnit.SECONDS);

        assertEquals(1, successCount.get());
        assertEquals(threadCount - 1, failCount.get());
    }
}
```
