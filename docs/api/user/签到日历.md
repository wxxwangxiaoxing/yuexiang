#### 核心位图解析

```java
/**
 * 从 tb_sign.sign_bitmap 解析已签到的日期列表
 * bit0→1号, bit1→2号, ..., bit30→31号
 */
public List<Integer> parseSignedDays(int signBitmap, int totalDaysInMonth) {
    List<Integer> days = new ArrayList<>();
    for (int day = 1; day <= totalDaysInMonth; day++) {
        if ((signBitmap >>> (day - 1) & 1) == 1) {
            days.add(day);
        }
    }
    return days;
}
```

#### 里程碑状态判定

```java
/**
 * 判定每个里程碑的状态
 * @param continuousDays  当前连续签到天数
 * @param claimedRuleIds  本周期已领取的 ruleId 集合
 */
private int determineMilestoneStatus(int requiredDays, Long ruleId,
                                     int continuousDays,
                                     Set<Long> claimedRuleIds) {
    if (claimedRuleIds.contains(ruleId)) return 2;      // 已领取
    if (continuousDays >= requiredDays) return 1;        // 可领取
    return 0;                                            // 未达成
}
```

#### nextMilestone 计算

```java
/**
 * 找到下一个「未达成且未领取」的里程碑
 */
private NextMilestoneVO findNextMilestone(List<TbSignRewardRule> rules,
                                          int continuousDays,
                                          Set<Long> claimedRuleIds) {
    return rules.stream()
        .filter(r -> r.getStatus() == 1) // 启用
        .filter(r -> !claimedRuleIds.contains(r.getId()))
        .filter(r -> continuousDays < r.getRequiredDays())
        .min(Comparator.comparingInt(TbSignRewardRule::getRequiredDays))
        .map(r -> NextMilestoneVO.builder()
            .targetDays(r.getRequiredDays())
            .remainDays(r.getRequiredDays() - continuousDays)
            .rewardName(r.getRewardName())
            .rewardIcon(r.getRewardIcon())
            .build())
        .orElse(null);
}
```

#### 后端核心逻辑

```java
private static final ZoneId ZONE = ZoneId.of("Asia/Shanghai");

@Transactional(rollbackFor = Exception.class)
public SignResultVO doSign(Long userId) {
    LocalDate today = LocalDate.now(ZONE);
    int year  = today.getYear();
    int month = today.getMonthValue();
    int day   = today.getDayOfMonth();

    // ═══ 0. 分布式锁 (防并发重复签到) ═══
    String lockKey = "lock:sign:" + userId;
    boolean locked = redisLockUtil.tryLock(lockKey, 5, TimeUnit.SECONDS);
    if (!locked) {
        throw new BizException(50001, "操作过于频繁，请稍后重试");
    }

    try {
        // ═══ 1. 获取或初始化月度签到记录 ═══
        TbSign sign = signMapper.selectByUserMonth(userId, year, month);
        if (sign == null) {
            sign = new TbSign();
            sign.setUserId(userId);
            sign.setYear(year);
            sign.setMonth(month);
            sign.setSignBitmap(0);
            sign.setSignDays(0);
            signMapper.insert(sign);
        }

        // ═══ 2. 位图判重 ═══
        int bitmap = sign.getSignBitmap();
        if ((bitmap >>> (day - 1) & 1) == 1) {
            throw new BizException(40020, "今日已签到，明天再来吧");
        }

        // ═══ 3. 计算连续签到天数(昨日往前 + 今日) ═══
        int continuousDays = calcContinuousDaysEndingAt(userId, today.minusDays(1)) + 1;

        // ═══ 4. 计算积分 ═══
        int basePoints  = getBasePoints(continuousDays);
        int bonusPoints = getBonusPoints(continuousDays);
        int totalReward = basePoints + bonusPoints;

        // ═══ 5. 更新签到位图(原子操作) ═══
        int newBitmap = bitmap | (1 << (day - 1));
        int newSignDays = sign.getSignDays() + 1;
        signMapper.updateBitmap(sign.getId(), newBitmap, newSignDays);

        // ═══ 6. 增加用户积分 ═══
        userInfoMapper.addPoints(userId, totalReward);
        int userTotalPoints = userInfoMapper.getPoints(userId);

        // ═══ 7. 更新历史最长连续天数 ═══
        userInfoMapper.updateMaxContinuousIfGreater(userId, continuousDays);

        // ═══ 8. 写入积分流水 ═══
        TbPointsRecord record = new TbPointsRecord();
        record.setUserId(userId);
        record.setType(1);                                 // 签到类型
        record.setPoints(totalReward);
        record.setBalance(userTotalPoints);
        record.setBizType("SIGN");
        record.setBizId(sign.getId());
        record.setDescription(buildSignDesc(basePoints, bonusPoints, continuousDays));
        pointsRecordMapper.insert(record);

        // ═══ 9. 检查里程碑解锁(本次是否新解锁了某个里程碑) ═══
        String cycleMonth = String.format("%d-%02d", year, month);
        MilestoneVO unlockedMilestone = checkNewMilestoneUnlock(
            userId, continuousDays, cycleMonth);

        // ═══ 10. 更新 Redis 缓存 ═══
        updateSignCache(userId, year, month, newBitmap, continuousDays);

        // ═══ 11. 判定动画类型 ═══
        String animation = "normal";
        if (unlockedMilestone != null) {
            animation = "milestone";
        } else if (newSignDays == today.lengthOfMonth()) {
            animation = "month_complete";
        }

        // ═══ 12. 构建返回 ═══
        return SignResultVO.builder()
            .signDate(today.toString())
            .day(day)
            .basePoints(basePoints)
            .bonusPoints(bonusPoints)
            .totalRewardPoints(totalReward)
            .userTotalPoints(userTotalPoints)
            .continuousDays(continuousDays)
            .isBonus(bonusPoints > 0)
            .bonusDesc(bonusPoints > 0
                ? String.format("🎉 连续签到%d天！额外奖励%d积分", continuousDays, bonusPoints)
                : null)
            .unlockedMilestone(unlockedMilestone)
            .animation(animation)
            .build();

    } finally {
        redisLockUtil.unlock(lockKey);
    }
}
```

#### 连续签到天数计算（跨月安全）

```java
/**
 * 计算截至 endDate 当天（含）的连续签到天数
 * 从 endDate 开始往前逐日检查，遇到未签到即终止
 *
 * @param userId  用户ID
 * @param endDate 检查的结束日期（含）
 * @return 连续签到天数
 */
private int calcContinuousDaysEndingAt(Long userId, LocalDate endDate) {
    int continuous = 0;
    LocalDate checkDate = endDate;

    // 缓存当前查询月份的签到数据，避免同月重复查库
    int cachedYear = -1, cachedMonth = -1;
    Integer cachedBitmap = null;

    while (true) {
        int y = checkDate.getYear();
        int m = checkDate.getMonthValue();
        int d = checkDate.getDayOfMonth();

        // 月份切换时重新加载位图
        if (y != cachedYear || m != cachedMonth) {
            // 优先从 Redis 读取，miss 则查库
            cachedBitmap = getSignBitmap(userId, y, m);
            cachedYear  = y;
            cachedMonth = m;
        }

        // 无记录 → 链条断裂
        if (cachedBitmap == null) break;

        // 检查该天 bit 位
        if ((cachedBitmap >>> (d - 1) & 1) == 1) {
            continuous++;
            checkDate = checkDate.minusDays(1);
        } else {
            break;
        }

        // 安全上限：回溯不超过 365 天
        if (continuous >= 365) break;
    }

    return continuous;
}

/**
 * 获取签到位图 (Redis → DB 两级读取)
 */
private Integer getSignBitmap(Long userId, int year, int month) {
    String cacheKey = String.format("sign:bitmap:%d:%d:%d", userId, year, month);
    Integer bitmap = (Integer) redisTemplate.opsForValue().get(cacheKey);

    if (bitmap != null) return bitmap;

    TbSign sign = signMapper.selectByUserMonth(userId, year, month);
    if (sign == null) return null;

    // 回填缓存，TTL = 月末 + 7 天
    LocalDate monthEnd = LocalDate.of(year, month, 1)
        .plusMonths(1).plusDays(6);
    long ttlSeconds = Duration.between(
        LocalDateTime.now(ZONE), monthEnd.atStartOfDay()).getSeconds();
    redisTemplate.opsForValue().set(cacheKey, sign.getSignBitmap(),
        Math.max(ttlSeconds, 86400), TimeUnit.SECONDS);

    return sign.getSignBitmap();
}
```

```java
@Transactional(rollbackFor = Exception.class)
public RepairResultVO repairSign(Long userId, int year, int month, int day) {
    LocalDate repairDate = LocalDate.of(year, month, day);
    LocalDate today = LocalDate.now(ZONE);

    // ═══ 0. 分布式锁 ═══
    String lockKey = "lock:sign:repair:" + userId;
    boolean locked = redisLockUtil.tryLock(lockKey, 5, TimeUnit.SECONDS);
    if (!locked) throw new BizException(50001, "操作频繁，请稍后重试");

    try {
        // ═══ 1. 日期校验 ═══
        if (!repairDate.isBefore(today)) {
            throw new BizException(40024, "不能补签今天或未来日期");
        }

        int windowDays = configService.getInt("sign.repair.window_days", 7);
        if (ChronoUnit.DAYS.between(repairDate, today) > windowDays) {
            throw new BizException(40023,
                String.format("仅支持补签最近%d天内的日期", windowDays));
        }

        // ═══ 2. 位图判重 ═══
        TbSign sign = signMapper.selectByUserMonth(userId,
            repairDate.getYear(), repairDate.getMonthValue());
        if (sign != null && (sign.getSignBitmap() >>> (day - 1) & 1) == 1) {
            throw new BizException(40021, "该日期已签到，无需补签");
        }

        // ═══ 3. 月度补签次数校验 ═══
        int maxRepair = configService.getInt("sign.repair.max_per_month", 3);
        int usedRepair = repairRecordMapper.countByUserMonth(
            userId, today.getYear(), today.getMonthValue());
        if (usedRepair >= maxRepair) {
            throw new BizException(40025,
                String.format("本月补签次数已达上限（%d次）", maxRepair));
        }

        // ═══ 4. 积分校验 & 扣减 ═══
        int costPoints = configService.getInt("sign.repair.cost_points", 50);
        int currentPoints = userInfoMapper.getPoints(userId);
        if (currentPoints < costPoints) {
            throw new BizException(40022,
                String.format("积分不足，补签需消耗%d积分，当前仅%d积分",
                    costPoints, currentPoints));
        }
        userInfoMapper.addPoints(userId, -costPoints);
        int remainPoints = currentPoints - costPoints;

        // ═══ 5. 更新目标月的签到位图 ═══
        int rYear  = repairDate.getYear();
        int rMonth = repairDate.getMonthValue();
        if (sign == null) {
            sign = new TbSign();
            sign.setUserId(userId);
            sign.setYear(rYear);
            sign.setMonth(rMonth);
            sign.setSignBitmap(0);
            sign.setSignDays(0);
            signMapper.insert(sign);
        }
        int newBitmap = sign.getSignBitmap() | (1 << (day - 1));
        signMapper.updateBitmap(sign.getId(), newBitmap, sign.getSignDays() + 1);

        // ═══ 6. 记录补签明细 ═══
        TbSignRepairRecord repair = new TbSignRepairRecord();
        repair.setUserId(userId);
        repair.setRepairDate(repairDate);
        repair.setCostPoints(costPoints);
        repairRecordMapper.insert(repair);

        // ═══ 7. 写入积分扣减流水 ═══
        pointsRecordMapper.insert(TbPointsRecord.builder()
            .userId(userId)
            .type(7)                                    // 补签扣减
            .points(-costPoints)
            .balance(remainPoints)
            .bizType("SIGN_REPAIR")
            .bizId(repair.getId())
            .description(String.format("补签%s消耗%d积分", repairDate, costPoints))
            .build());

        // ═══ 8. 重新计算连续天数 ═══
        int prevContinuous = calcContinuousDaysEndingAt(userId, today.minusDays(1));
        boolean isSignedToday = isSignedToday(userId, today);
        int newContinuous = isSignedToday
            ? calcContinuousDaysEndingAt(userId, today)
            : prevContinuous;  // 今天还没签到，连续天数截至昨天

        // 更新历史最长
        userInfoMapper.updateMaxContinuousIfGreater(userId, newContinuous);

        // ═══ 9. 检查里程碑解锁 ═══
        String cycleMonth = String.format("%d-%02d",
            today.getYear(), today.getMonthValue());
        MilestoneVO milestone = null;
        // 补签可能使连续天数跳变，需检查所有可能新达成的里程碑
        if (newContinuous > prevContinuous) {
            milestone = checkMilestonesInRange(userId,
                prevContinuous + 1, newContinuous, cycleMonth);
        }

        // ═══ 10. 清除/更新缓存 ═══
        evictSignCache(userId, rYear, rMonth);
        updateContinuousCache(userId, newContinuous);

        return RepairResultVO.builder()
            .signDate(repairDate.toString())
            .costPoints(costPoints)
            .remainPoints(remainPoints)
            .continuousDays(newContinuous)
            .continuousChanged(newContinuous != prevContinuous)
            .unlockedMilestone(milestone)
            .build();

    } finally {
        redisLockUtil.unlock(lockKey);
    }
}

/**
 * 检查连续天数在 [from, to] 范围内是否有新可解锁的里程碑
 * (补签可能一次修复多天链条，跨过多个里程碑阈值)
 */
private MilestoneVO checkMilestonesInRange(Long userId, int from, int to,
                                            String cycleMonth) {
    List<TbSignRewardRule> rules = rewardRuleMapper.selectAllEnabled();
    Set<Long> claimed = rewardRecordMapper.selectClaimedRuleIds(userId, cycleMonth);

    // 返回最高的那个新达成里程碑(给用户最大惊喜)
    return rules.stream()
        .filter(r -> r.getRequiredDays() >= from && r.getRequiredDays() <= to)
        .filter(r -> !claimed.contains(r.getId()))
        .max(Comparator.comparingInt(TbSignRewardRule::getRequiredDays))
        .map(r -> MilestoneVO.builder()
            .ruleId(r.getId())
            .requiredDays(r.getRequiredDays())
            .rewardName(r.getRewardName())
            .rewardIcon(r.getRewardIcon())
            .message(String.format("补签修复了连续签到链！解锁「连续%d天」成就",
                r.getRequiredDays()))
            .build())
        .orElse(null);
}
```

```java
        // ═══ 2. 检查本周期是否已领取 ═══
        LocalDate today = LocalDate.now(ZONE);
        String cycleMonth = String.format("%d-%02d",
            today.getYear(), today.getMonthValue());

        boolean claimed = rewardRecordMapper.existsByUserRuleCycle(
            userId, ruleId, cycleMonth);
        if (claimed) {
            throw new BizException(40031, "该奖励已领取，不可重复领取");
        }

        // ═══ 3. 校验连续签到天数 ═══
        int continuousDays = calcContinuousDaysEndingAt(userId, today);
        if (continuousDays < rule.getRequiredDays()) {
            throw new BizException(40030,
                String.format("连续签到天数未达要求（需连续%d天，当前%d天）",
                    rule.getRequiredDays(), continuousDays));
        }

        // ═══ 4. 发放奖励 ═══
        Object rewardDetail;
        Long bizId = null;

        switch (rule.getRewardType()) {
            case 1: // 抽奖机会
                lotteryService.grantChance(userId, rule.getRewardValue());
                rewardDetail = Map.of(
                    "lotteryChances", rule.getRewardValue(),
                    "lotteryUrl", "/pages/lottery/index"
                );
                break;

            case 2: // 优惠券
                VoucherGrantResult vr = voucherService.grantSignRewardVoucher(
                    userId, rule.getVoucherId());
                bizId = vr.getOrderId();
                rewardDetail = Map.of(
                    "voucherId", vr.getVoucherId(),
                    "voucherTitle", vr.getTitle(),
                    "validBeginTime", vr.getBeginTime(),
                    "validEndTime", vr.getEndTime()
                );
                break;

            case 3: // 积分
                userInfoMapper.addPoints(userId, rule.getRewardValue());
                int newPoints = userInfoMapper.getPoints(userId);
                pointsRecordMapper.insert(TbPointsRecord.builder()
                    .userId(userId).type(4).points(rule.getRewardValue())
                    .balance(newPoints).bizType("SIGN_MILESTONE").bizId(ruleId)
                    .description("签到里程碑奖励+" + rule.getRewardValue() + "积分")
                    .build());
                rewardDetail = Map.of(
                    "points", rule.getRewardValue(),
                    "totalPoints", newPoints
                );
                break;

            default:
                throw new BizException(50000, "未知奖励类型");
        }

        // ═══ 5. 写入领取记录 ═══
        TbSignRewardRecord record = new TbSignRewardRecord();
        record.setUserId(userId);
        record.setRuleId(ruleId);
        record.setCycleMonth(cycleMonth);
        record.setRewardType(rule.getRewardType());
        record.setRewardValue(rule.getRewardValue());
        record.setStatus(1);
        record.setBizId(bizId);
        rewardRecordMapper.insert(record);

        // ═══ 6. 构建返回 ═══
        String message = buildClaimMessage(rule);
        return ClaimResultVO.builder()
            .ruleId(ruleId)
            .rewardType(rule.getRewardType())
            .rewardName(rule.getRewardName())
            .rewardDetail(rewardDetail)
            .message(message)
            .build();

    } finally {
        redisLockUtil.unlock(lockKey);
    }
}

private String buildClaimMessage(TbSignRewardRule rule) {
    switch (rule.getRewardType()) {
        case 1: return String.format("🎰 获得%d次抽奖机会，快去试试手气吧！",
                    rule.getRewardValue());
        case 2: return String.format("🎉 %s已发放至您的账户，请在「我的优惠券」中查看",
                    rule.getRewardName());
        case 3: return String.format("🎉 %d积分已到账！", rule.getRewardValue());
        default: return "🎁 奖励已发放！";
    }
}
```

```java
/**
 * 签到后更新缓存
 */
private void updateSignCache(Long userId, int year, int month,
                              int newBitmap, int continuousDays) {
    String bitmapKey = String.format("sign:bitmap:%d:%d:%d", userId, year, month);
    String contKey   = String.format("sign:continuous:%d", userId);

    // 位图缓存
    LocalDate monthEnd = LocalDate.of(year, month, 1).plusMonths(1).plusDays(6);
    long ttl = Duration.between(LocalDateTime.now(ZONE),
        monthEnd.atStartOfDay()).getSeconds();
    redisTemplate.opsForValue().set(bitmapKey, newBitmap,
        Math.max(ttl, 86400), TimeUnit.SECONDS);

    // 连续天数缓存
    redisTemplate.opsForValue().set(contKey, continuousDays,
        48, TimeUnit.HOURS);
}

/**
 * 补签后清除目标月缓存(因为位图被修改)
 */
private void evictSignCache(Long userId, int year, int month) {
    String bitmapKey = String.format("sign:bitmap:%d:%d:%d", userId, year, month);
    redisTemplate.delete(bitmapKey);
}
```

## 八、边界场景与处理

| #  | 场景              | 处理方案                                                           |
| -- | --------------- | -------------------------------------------------------------- |
| 1  | **跨月签到计算**      | `calcContinuousDaysEndingAt` 自动跨月回溯位图，无缝衔接                     |
| 2  | **跨年签到**        | 年份在查询参数中体现，位图按 `(userId, year, month)` 唯一索引                    |
| 3  | **23:59:59 签到** | 入口层固定 `LocalDate today = LocalDate.now(ZONE)`，整个事务使用同一 `today` |
| 4  | **并发双击**        | 分布式锁 `lock:sign:{userId}` + 位图判重双保险                            |
| 5  | **补签目标月无记录**    | 自动创建空位图记录 `sign_bitmap=0, sign_days=0`                         |
| 6  | **补签修复连续链**     | 补签后重新计算连续天数，可能跨过多个里程碑阈值                                        |
| 7  | **里程碑重复领取**     | `UNIQUE KEY (user_id, rule_id, cycle_month)` + 分布式锁            |
| 8  | **月度里程碑重置**     | `cycle_month` 按自然月隔离，新月可重新领取                                   |
| 9  | **积分扣减为负**      | 先查再扣，`currentPoints < costPoints` 时拒绝                          |
| 10 | **用户时区差异**      | 统一服务端 UTC+8，`serverTime` 返回服务器时间供前端校准                          |
| 11 | **排行榜实时性**      | 定时任务 10 分钟刷新，接受短暂延迟，减少 DB 压力                                   |
| 12 | **历史月浏览**       | 仅展示已签到日期，签到按钮隐藏，补签按钮受 7 天窗口限制                                  |
